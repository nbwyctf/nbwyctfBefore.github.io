这部分主要实现界面的设置和监听鼠标点击事件。

public class Windows implements ActionListener {
    //  创建公共静态对象方便调用
    public static JFrame frame=new JFrame();
    public static JButton jButton=new JButton();

    public static JLabel label1=new JLabel("待开:"+UNOPEN);
    public static JLabel label2=new JLabel("已开:"+OPENED);
    public static JLabel label3=new JLabel("剩余雷数:"+(MINECOUNT-flagCount));
    public static JLabel label4=new JLabel("用时:"+second+"s");

    //  定义计时器
    public Timer timer=new Timer(1000,this);
    //  创建实例对象
    Function function=new Function();

    // 创建窗口
    public void createWindows(){
        //定义窗口大小（600*700）
        frame.setSize(800,800);
        //窗口是否可以改变大小
        frame.setResizable(false);
        //点击可关闭
        frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        //设置窗口布局
        frame.setLayout(new BorderLayout());
        //显示窗口
        frame.setVisible(true);
    }
    // 创建的按钮放窗口顶部
    public void setHeader(){
        //创建布局
        JPanel panel=new JPanel(new GridBagLayout());
        GridBagConstraints c1=new GridBagConstraints(0,0,4,1,1.0,1.0,GridBagConstraints.CENTER,GridBagConstraints.BOTH,new Insets(0,0,0,0),0,0);
        panel.add(jButton,c1);

        //设置透明度
        label1.setOpaque(true);
        //设置背景色
        label1.setBackground(Color.white);
        //设置边框颜色
        label1.setBorder(BorderFactory.createLineBorder(Color.LIGHT_GRAY));
        //label2、label3同理
        label2.setOpaque(true);
        label2.setBackground(Color.white);
        label2.setBorder(BorderFactory.createLineBorder(Color.LIGHT_GRAY));
        label3.setOpaque(true);
        label3.setBackground(Color.white);
        label3.setBorder(BorderFactory.createLineBorder(Color.LIGHT_GRAY));
        label4.setOpaque(true);
        label4.setBackground(Color.white);
        label4.setBorder(BorderFactory.createLineBorder(Color.LIGHT_GRAY));
        //设置三个状态显示栏
        GridBagConstraints c2=new GridBagConstraints(0,1,1,1,1.0,1.0,GridBagConstraints.CENTER,GridBagConstraints.BOTH,new Insets(0,0,0,0),0,0);
        panel.add(label1,c2);
        GridBagConstraints c3=new GridBagConstraints(1,1,1,1,1.0,1.0,GridBagConstraints.CENTER,GridBagConstraints.BOTH,new Insets(0,0,0,0),0,0);
        panel.add(label2,c3);
        GridBagConstraints c4=new GridBagConstraints(2,1,1,1,1.0,1.0,GridBagConstraints.CENTER,GridBagConstraints.BOTH,new Insets(0,0,0,0),0,0);
        panel.add(label3,c4);
        GridBagConstraints c5=new GridBagConstraints(3,1,1,1,1.0,1.0,GridBagConstraints.CENTER,GridBagConstraints.BOTH,new Insets(0,0,0,0),0,0);
        panel.add(label4,c5);
        //添加至布局的上方
        frame.add(panel, BorderLayout.NORTH);
    }
    //设置按钮
    public void setButtons(){
        //创建容器
        Container container=new Container();
        //创建布局
        container.setLayout(new GridLayout(ROW,COL));
        //添加至整体布局的中央
        frame.add(container,BorderLayout.CENTER);
        //创建按钮
        for(int i=0;i<ROW;i++){
            for(int j=0;j<COL;j++){
                JButton btn=new JButton();
                btn.setOpaque(true);
                btn.setBackground(Color.yellow);
                btn.addMouseListener(new MouseEventListener());
                container.add(btn);
                buttons[i][j]=btn;
            }
        }
    }

    //编写btn监听器工作
    @Override
    public void actionPerformed(ActionEvent e) {
        //用于计时
        if(e.getSource() instanceof Timer){
            second++;
            label3.setText("用时"+second+"s");
            timer.start();
        }
    }

    private class MouseEventListener extends MouseAdapter {
        @Override
        public void mouseClicked(MouseEvent e) {
            super.mouseClicked(e);
            //按键响应
            JButton btn = (JButton) e.getSource();
            //监听鼠标左键
            if (e.getButton() == MouseEvent.BUTTON1) {
                //游戏功能判定
                for (int i = 0; i < ROW; i++)
                {
                    for (int j = 0; j < COL; j++)
                    {
                        if(btn.equals(buttons[i][j]))
                        {
                            if (data[i][j] == MINECODE && flag[i][j] == 0) {
                                function.lose();
                            }else{
                                if(flag[i][j] == 0){function.openCell(i, j);}
                                //如果这个格子为0且被开过
                                if(data[i][j]==0 && !buttons[i][j].isEnabled()){ function.openNum(); }
                                //检测是否胜利
                                function.checkWin();
                            }
                        }

                    }
                }

            }

            //监听鼠标右键
            if (e.getButton() == MouseEvent.BUTTON3) {
                //游戏功能判定
                for (int i = 0; i < ROW; i++)
                {
                    for (int j = 0; j < COL; j++)
                    {
                        if(btn.equals(buttons[i][j]))
                        {
                            //插上旗帜
                            function.setFlag(i,j);
                        }

                    }
                }
            }
        }
    }
}

Function

这部分实现添加雷，响应鼠标事件，连开逻辑，和判断输赢等功能

public class Function implements ActionListener {
    public Timer timer = new Timer(1000, this);

    //添加雷
    public void add(){
        Random random = new Random();
        for (int i = 0; i < MINECOUNT;) {
            //使用随机数生成雷
            int r = random.nextInt(ROW);
            int c = random.nextInt(COL);
            //避免生成位置重复
            if (data[r][c] != MINECODE) {
                data[r][c] = MINECODE;
                i++;
            }
        }
        //计算周围雷的数量
        for (int i = 0; i < ROW; i++) {
            for (int j = 0; j < COL; j++) {
                flag[i][j] = 0;  //初始化二维数组flag为全0
                if (data[i][j] == MINECODE) continue;
                int num = 0;//初始化格子周围的雷数
                if (i>0 && j>0 && data[i - 1][j - 1] == MINECODE) { num++; }
                if (i>0 && data[i - 1][j] == MINECODE){num++;}
                if (i>0 && j <COL-1 && data[i - 1][j + 1] == MINECODE){num++;}
                if (j>0 && data[i][j - 1] == MINECODE){num++;}
                if (j<COL-1 && data[i][j + 1] == MINECODE){num++;}
                if (i<ROW-1 && j<COL-1 && data[i + 1][j + 1] == MINECODE) { num++; }
                if (i<ROW-1 && data[i+1][j] == MINECODE){num++;}
                if (i<ROW-1 && j>0 && data[i+1][j-1] == MINECODE){num++;}
                data[i][j] = num;
            }
        }
    }

    public void restart() {
        //重置格子
        for(int i=0;i<ROW;i++){
            for(int j=0;j<COL;j++){
                data[i][j]=0;
                buttons[i][j].setBackground(Color.yellow);
                buttons[i][j].setEnabled(true);
                buttons[i][j].setText("");
            }
        }
        //重置参数
        UNOPEN=ROW*COL;
        OPENED=0;
        second=0;
        flagCount=0;
        //重置状态栏
        Windows.label1.setText("待开:"+UNOPEN);
        Windows.label2.setText("已开:"+OPENED);
        Windows.label3.setText("剩余雷数:"+(MINECOUNT-flagCount));
        Windows.label4.setText("用时:"+second+"s");
        //重新添加地雷
        add();
        //重新启动计时器
        timer.start();
    }

    public void lose() {
        for (int i = 0; i < ROW; i++) {
            for (int j = 0; j < COL; j++) {
                JButton btn = buttons[i][j];
                btn.setEnabled(false);
                if(data[i][j] == MINECODE){
                    btn.setOpaque(true);
                    btn.setBackground(Color.red);
                }
            }
        }
        timer.stop();
        int choice = JOptionPane.showConfirmDialog(null,"你踩雷了,是否再来一局","选项",JOptionPane.YES_NO_OPTION);
        if (choice == JOptionPane.YES_OPTION) {
            restart();
        }
    }

    //打开格子
    public void openCell(int i, int j) {
        JButton btn = buttons[i][j];
        //去除已经打开的格子
        if (!btn.isEnabled()) {
            return;
        }

        /*
        if (flag[i][j] == 1) {
            flag[i][j] = 2;
            btn.setText("?");
            flagCount--;
            Windows.label3.setText("剩余雷数:"+(MINECOUNT-flagCount));
            return;
        }
        if (flag[i][j] == 2) {
            flag[i][j] = 0;
            btn.setText("");
            return;
        }
        */

        //设置打开格子的效果
        btn.setEnabled(false);
        btn.setOpaque(true); //设置透明
        if(data[i][j]!=0){
            //非空白格
            btn.setText(data[i][j] + "");
            btn.setBackground(Color.green);
        }else{
            //空白格
            btn.setBackground(Color.white);
        }

        //调用方法，更新label中的值
        updateCount();
        //使用递归的思想，实现级联打开
        if (data[i][j] == 0) {
            if(i>0 && j>0 && data[i-1][j-1]==0)
                openCell(i-1,j-1);
            if(i>0 && data[i-1][j]==0)
                openCell(i-1,j);
            if(i>0 && j<COL-1 && data[i-1][j+1]==0)
                openCell(i-1,j+1);
            if(j>0 && data[i][j-1]==0)
                openCell(i,j-1);
            if(j<COL-1 && data[i][j+1]==0)
                openCell(i,j+1);
            if(i<ROW-1 && j>0 && data[i+1][j-1]==0)
                openCell(i+1,j-1);
            if(i<ROW-1 && data[i+1][j]==0)
                openCell(i+1,j);
            if(i<ROW-1 && j<COL-1 &&data[i+1][j+1]==0)
                openCell(i+1,j+1);
        }
    }

    //打开空白格旁边的数字格
    public void openNum() {
        for (int i = 0; i < ROW; i++) {
            for (int j = 0; j < COL; j++) {
                //如果这个格子为0且被开过
                if (data[i][j] == 0 && !buttons[i][j].isEnabled()) {
                    //只要这个格子周围的格子不为0且未被翻开过，就翻开
                    if(i>0 && j>0 && data[i-1][j-1]!=0 && buttons[i-1][j-1].isEnabled())
                        showNum(i-1,j-1);
                    if(i>0 && data[i-1][j]!=0 && buttons[i-1][j].isEnabled())
                        showNum(i-1,j);
                    if(i>0 && j<COL-1 && data[i-1][j+1]!=0 && buttons[i-1][j+1].isEnabled())
                        showNum(i-1,j+1);
                    if(j>0 && data[i][j-1]!=0 && buttons[i][j-1].isEnabled())
                        showNum(i,j-1);
                    if(j<COL-1 && data[i][j+1]!=0 && buttons[i][j+1].isEnabled())
                        showNum(i,j+1);
                    if(i<ROW-1 && j>0 && data[i+1][j-1]!=0 && buttons[i+1][j-1].isEnabled())
                        showNum(i+1,j-1);
                    if(i<ROW-1 && data[i+1][j]!=0 && buttons[i+1][j].isEnabled())
                        showNum(i+1,j);
                    if(i<ROW-1 && j<COL-1 &&data[i+1][j+1]!=0 && buttons[i+1][j+1].isEnabled())
                        showNum(i+1,j+1);
                }
            }
        }
    }

    public void showNum(int i, int j){
        JButton btn = buttons[i][j];
        btn.setEnabled(false);
        btn.setOpaque(true); //设置透明

        btn.setText(data[i][j] + "");
        btn.setBackground(Color.green);
        updateCount();
    }

    //插上旗帜
    public void setFlag(int i, int j) {
        JButton btn = buttons[i][j];
        //去除已经打开的格子
        if (!btn.isEnabled()) {
            return;
        }
        if (flag[i][j] == 1) {
            flag[i][j] = 2;
            btn.setText("?");
            flagCount--;
            Windows.label3.setText("剩余雷数:"+(MINECOUNT-flagCount));
            return;
        }
        if (flag[i][j] == 2) {
            flag[i][j] = 0;
            btn.setText("");
            return;
        }
        //该格子flag为空
        flag[i][j] = 1;
        btn.setText("🔺");
        if((MINECOUNT-flagCount)>0){
            flagCount++;
        }
        Windows.label3.setText("剩余雷数:"+(MINECOUNT-flagCount));

    }

    //更新label的值
    private void updateCount() {
        OPENED ++;
        UNOPEN --;
        Windows.label1.setText("待开：" + UNOPEN);
        Windows.label2.setText("已开：" + OPENED);
    }

    public void checkWin() {
        int count = 0;
        for (int i = 0; i < ROW; i++) {
            for (int j = 0; j < COL; j++) {
                if(buttons[i][j].isEnabled())count++;
            }
        }
        if (count == MINECOUNT) {
            timer.stop();
            int choice = JOptionPane.showConfirmDialog(null,"你获胜了,是否再来一局","选项",JOptionPane.YES_NO_OPTION);
            if (choice == JOptionPane.YES_OPTION) {
                restart();
            }
        }
    }

    //编写监听器，用于计时
    @Override
    public void actionPerformed(ActionEvent e) {
        if (e.getSource() instanceof Timer) {
            second++;
            Windows.label4.setText("用时"+second+"s");
            timer.start();

        }
    }
}

StartMineSweepingGame

这部分是主函数的实现和初始化参数

public class StartMineSweepingGame implements ActionListener {
    public Timer timer = new Timer(1000, this);
    Function function = new Function();
    Windows windows = new Windows();
    //增加第一次点击不能是雷功能=》如果是雷则重开并打开这个格子直至这个格子不在是雷

    // 定义地图大小及数据结构
    public static int ROW = 16;
    public static int COL = 16;
    public static int flagCount = 0;
    public static int[][] data = new int[ROW][COL];
    public static JButton[][] buttons = new JButton[ROW][COL];
    public static int[][] flag = new int[ROW][COL]; //在二维数组中的确定旗帜坐标
    // 定义地雷的个数
    public static int MINECOUNT = 10; //40 16*16
    // 定义表示地雷的特征数
    public static int MINECODE = -1;
    // 定义状态栏参数
    public static int UNOPEN = ROW * COL;
    public static int OPENED = 0;
    public static int second = 0;

    public StartMineSweepingGame(){
        //调用创建窗口
        windows.createWindows();
        //创建顶部状态框
        windows.setHeader();
        //设置按钮
        windows.setButtons();
        //创建地雷
        function.add();
        function.restart();
    }

    public static void main(String[] args) {
        new StartMineSweepingGame();
    }

    @Override
    public void actionPerformed(ActionEvent e) {
        if (e.getSource() instanceof Timer) {
            second++;
            Windows.label3.setText("用时" + second + "s");
            timer.start();
        }
    }
}
