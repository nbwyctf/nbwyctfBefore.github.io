    ascii_letters常量

    string.ascii_letters

    参数说明：返回26个大小写英文字母

    获取26个大小写英文字母

    import string
chars = string.ascii_letters
print(chars)
#abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ

    随机生存5组大小些字母组成的字符串

    import string,random
for i in range(5):
    #sample()函数从指定序列中随机截取8个字符
    myMode="".join(random.sample(string.ascii_letters,8))
    print(myMode)
'''
XnkieIJK
klGwpWAr
WSiNjfGp
ZeDfNGlv
xYDSypbB
'''

    随机生成100个激活码

    import random,string
def IDstr(num,len = 8):
    f=open("./mycode.txt",'w')
    for i in range(num):
        chars = string.ascii_letters+string.digits
        #choice()函数从任何序列中随机选取n位字符
        s =[random.choice(chars) for i in range(len)]
        f.write("{0}\n".format("".join(s)))   
    f.close()
#生成100个激活码
IDstr(100)

    capwords()方法

    将单词首字母转换为大写

    string.capwords(s,sep=None)

    参数说明：

    • s: 必选参数，用于指定要转换的字符串
    • sep:可选参数，为空或None,多个空格胡ibei一个空格代替，字符串开头和结尾的空格会被移除；另外，sep这各参数是用来分割和组合字符串的
    • 返回值：返回转换后的字符串

    将所有单词的首写字母转换为大写

    import string
en = "there's nothing you can do that can't be done"
result = string.capwords(en)
print(result)
#There's Nothing You Can Do That Can't Be Done

    用分隔符仅将指定字母转换为大写形式

    import string
en = 'do not.for one repulse.give up the purpose that you resolved to effect'
result = string.capwords(en,sep='.')#以.为分隔符
print(result)
#Do not.For one repulse.Give up the purpose that you resolved to effect

    Formatter()类

    format()方法

    参数说明：

    format(format_string,*args,**kwargs)

    • format_string: 表示需要格式化的目标的目标字符串
    • *args:表示填充目标植复仇的序列，是字典或元组
    • **kwargs:关键字参数
    • 返回值：返回格式化后的字符串

    生成数据编号

    import string
formatter = string.Formatter()
s1 = '{:0>3}'
s2 = '{:0>5}'
s3 = 'a{:0>6}'
print(formatter.format(s1,1))
print(formatter.format(s2,'08'))
print(formatter.format(s3,886))
#001
#00008
#a000886

    格式化百分比

    import string
formatter = string.Formatter()
s1 = '{:%}'#将小数格式化成百分比
s2 = '{:.0%}'#将小数格式化后保留整数百分比
s3 = '{:.2%}'#将小数格式化后保留两位小数的百分比
s4 = '{:10.3%}'#将小数格式化后保留三位小数10位百分比
print(formatter.format(s1,0.1314))
print(formatter.format(s2,0.1314))
print(formatter.format(s3,0.1314))
print(formatter.format(s4,0.1314))
'''
13.140000%
13%
13.14%
   13.140%
'''

    格式化金额

    import string
formatter = string.Formatter()
s1 = '${:.2f}'#添加美元符号，小数保留两位
s2 = '￥{:,.2f}'#添加人民币符号，用千位分割符进行区分
s3 = '￡{:,.2f}'#添加人英镑符号，用千位分割符进行区分
s4 = '€{:.2f}'#添加欧元符号，小数保留两位
print(formatter.format(s1,0.88))
print(formatter.format(s2,888800))
print(formatter.format(s3,1234567))
print(formatter.format(s4,1234567))
'''
$0.88
￥888,800.00
￡1,234,567.00
€1234567.00
'''

    format_field()方法

    参数说明：

    format(value,format_spec)

    • value: 表示要格式化的对象
    • format_spec:格式化规范，字符串类型
    • 返回值：返回格式化后的字符串

    为字符串填充占位符

    import string
formatter = string.Formatter()
val = "电子科技大学"
str1 = formatter.format_field(val,'*^30')#利用*当占位符
print(str1)
str2 = formatter.format_field(val,'-^20')#利用-当占位符
print(str2)
'''
************电子科技大学************
-------电子科技大学-------
'''

    格式化数字(补0或补x)

    import string
formatter = string.Formatter()
val = 5
str1 = formatter.format_field(val,'0>2')#数字补0，填充左边，长度为2
print(str1)
str2 = formatter.format_field(val,'x<4')#数字补x，填充右边，长度为4
print(str2)
str3 = formatter.format_field(val,'x^4')#数字补x，填充左右，长度为3
print(str3)
'''
05
5xxx
x5xx
'''
