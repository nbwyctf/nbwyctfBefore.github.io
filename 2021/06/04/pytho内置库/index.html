String模块

ascii_letters常量

string.ascii_letters

参数说明：返回26个大小写英文字母

获取26个大小写英文字母

import string
chars = string.ascii_letters
print(chars)
#abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ

随机生存5组大小些字母组成的字符串

import string,random
for i in range(5):
    #sample()函数从指定序列中随机截取8个字符
    myMode="".join(random.sample(string.ascii_letters,8))
    print(myMode)
'''
XnkieIJK
klGwpWAr
WSiNjfGp
ZeDfNGlv
xYDSypbB
'''

随机生成100个激活码

import random,string
def IDstr(num,len = 8):
    f=open("./mycode.txt",'w')
    for i in range(num):
        chars = string.ascii_letters+string.digits
        #choice()函数从任何序列中随机选取n位字符
        s =[random.choice(chars) for i in range(len)]
        f.write("{0}\n".format("".join(s)))   
    f.close()
#生成100个激活码
IDstr(100)

ascii_lowercase常量

string.ascii_lowercase

参数说明：返回26个小写字母

获取26个小写字母

import string
chars = string.ascii_lowercase
print(chars)
#abcdefghijklmnopqrstuvwxyz

判断输入字母是否全为小写

from sympy import false
def check(value):
    for letter in value:
        if letter not in string.ascii_lowercase:
            return false
myval = input("请输入小写字母：")
print(check(myval))

生成由小写字母和数字随机生成的10位密码

import string, random
passwords = ''.join([random.choice(string.ascii_lowercase+string.digits) for n in range(10)])
print(passwords)
